00318 static inline __attribute__((always_inline))
00319 void badPinCheck(uint8_t pin) {
00320   if (!__builtin_constant_p(pin) || pin >= digitalPinCount) {
00321      badPinNumber();
00322   }
00323 }
00324 //------------------------------------------------------------------------------
00330 static inline __attribute__((always_inline))
00331 void fastBitWriteSafe(volatile uint8_t* address, uint8_t bit, bool level) {
00332   uint8_t oldSREG;
00333   if (address > (uint8_t*)0X5F) {
00334     oldSREG = SREG;
00335     cli();
00336   }
00337   if (level) {
00338     *address |= 1 << bit;
00339   } else {
00340     *address &= ~(1 << bit);
00341   }
00342   if (address > (uint8_t*)0X5F) {
00343     SREG = oldSREG;
00344   }
00345 }
00346 //------------------------------------------------------------------------------
00351 static inline __attribute__((always_inline))
00352 bool fastDigitalRead(uint8_t pin) {
00353   badPinCheck(pin);
00354   return (*pinMap[pin].pin >> pinMap[pin].bit) & 1;
00355 }
00356 //------------------------------------------------------------------------------
00363 static inline __attribute__((always_inline))
00364 void fastDigitalToggle(uint8_t pin) {
00365   badPinCheck(pin);
00366     if (pinMap[pin].pin > (uint8_t*)0X5F) {
00367       // must write bit to high address port
00368       *pinMap[pin].pin = 1 << pinMap[pin].bit;
00369     } else {
00370       // will compile to sbi and PIN register will not be read.
00371       *pinMap[pin].pin |= 1 << pinMap[pin].bit;
00372     }
00373 }
00374 //------------------------------------------------------------------------------
00379 static inline __attribute__((always_inline))
00380 void fastDigitalWrite(uint8_t pin, bool level) {
00381   badPinCheck(pin);
00382   fastBitWriteSafe(pinMap[pin].port, pinMap[pin].bit, level);
00383 }
00384 //------------------------------------------------------------------------------
00391 static inline __attribute__((always_inline))
00392 void fastPinMode(uint8_t pin, bool mode) {
00393   badPinCheck(pin);
00394   fastBitWriteSafe(pinMap[pin].ddr, pinMap[pin].bit, mode);
00395 }

00403 static inline __attribute__((always_inline))
00404 void fastPinConfig(uint8_t pin, bool mode, bool level) {
